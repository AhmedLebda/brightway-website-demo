Design Tokens (CSS)

Copy the complete token sheet into any project. All tokens are defined as CSS custom properties on :root.

CSS — :root tokens
:root { /* ── Brand ── */ --navy: #0D3B6E; --navy-dark: #071E3D; --navy-mid: #164A82; --teal: #1DB8A4; --teal-light: #E6F8F6; --teal-dark: #0E9485; --amber: #F4A623; --amber-light: #FEF3DD; /* ── Neutrals ── */ --bg: #F7F9FC; --white: #FFFFFF; --text: #0D1B2A; --muted: #6B7B8E; --subtle: #A0B0C0; --border: rgba(13,59,110,0.09); /* ── Semantic ── */ --success: #2DBE8B; --info: #4A9AF4; --warning: #F4A623; --error: #EF4444; /* ── Shadows ── */ --shadow-sm: 0 2px 12px rgba(13,59,110,0.07); --shadow: 0 8px 32px rgba(13,59,110,0.12); --shadow-lg: 0 24px 64px rgba(13,59,110,0.16); --shadow-teal: 0 8px 24px rgba(29,184,164,0.35); /* ── Radius ── */ --r: 8px; --r-md: 16px; --r-lg: 24px; --r-xl: 32px; /* ── Easing ── */ --ease: cubic-bezier(.25,.46,.45,.94); --spring: cubic-bezier(.34,1.3,.64,1); --spring-soft: cubic-bezier(.34,1.15,.64,1); --spring-lang: cubic-bezier(.34,1.4,.64,1); /* ── Duration ── */ --dur-fast: 150ms; --dur-normal: 300ms; --dur-reveal: 650ms; --dur-counter: 1500ms; /* ── Spacing ── */ --space-1: 4px; --space-2: 8px; --space-3: 12px; --space-4: 16px; --space-5: 20px; --space-6: 24px; --space-8: 32px; --space-10:40px; --space-12: 48px; --space-16:64px; --space-24: 96px; --section: 100px; /* ── Typography ── */ --font-display: 'Plus Jakarta Sans', sans-serif; --font-body: 'Inter', sans-serif; --font-mono: 'JetBrains Mono', monospace; }
